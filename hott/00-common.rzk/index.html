The type of logical equivalences between types

#def iff (A B : U) : U
  := prod (A -> B) (B -> A)

Basic function definitions

#section basic-functions

#variables A B C D : U

#def composition
  (g : B -> C)    -- The second function.
  (f : A -> B)    -- The first function.
  : A -> C        -- The composite function.
  := \z -> g (f z)

#def triple-composition
  (h : C -> D)
  (g : B -> C)
  (f : A -> B)
  : A -> D
  := \z -> h (g (f z))

#def identity
  : A -> A
  := \a -> a

#def constant
  (b : B)         -- The constant output value.
  : A -> B
  := \a -> b

#end basic-functions

Substitution

-- Reindexing a type family along a function into the base type.
#def reindex
  (A B : U)
  (f : B -> A)
  (C : A -> U)
  : (B -> U)
  := \b -> C (f b)

Final projection

-- Constant map into a pointed type
#def const
  (A : U)
  (B : U)
  (b : B)
  : (A -> B)
  := \a -> b

-- Final projection as a constant map
#def final-projection
  (A : U)
  : A -> Unit
  := const A Unit unit

Unit type

#def ind-Unit
  (C : Unit -> U)
  (C-unit : C unit)
  (x : Unit)
  : C x
  := C-unit

#def uniq-Unit
  (x : Unit)
  : x = unit
  := refl

#def isProp-Unit
  (x y : Unit)
  : x = y
  := refl
